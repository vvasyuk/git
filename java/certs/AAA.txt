set PATH=%PATH%;D:\work\installed\java\bin\
set PATH=%PATH%;D:\work\installed\OpenSSL\bin\


### generate a public-private key pair
keytool -genkey -keystore ca.jks -alias caKey -storepass password -keypass password -dname "CN=cn, OU=ou, O=o, L=l, S=s, C=UA"

### check jks
keytool -list -v -keystore ca.jks -storepass password
keytool -list -v -keystore truststore.jks -storepass password

### export cert from jks
keytool -export -keystore ca.jks -alias caKey -file ca.cer -storepass password

### print cert
keytool -printcert -file ca.cer

### create jks from cert
keytool -import -alias caKey -file ca.cer -keystore truststore.jks -storepass password

### create server jks
keytool -genkey -keystore server.jks -alias server -storepass password -keypass password -dname "CN=cn, OU=ou, O=o, L=l, S=s, C=UA"
