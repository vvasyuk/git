### generate a public-private key pair
D:\work\installed\java\bin\keytool -genkey -keystore ca.jks -alias caKey -storepass password -keypass password -dname "CN=cn, OU=ou, O=o, L=l, S=s, C=UA"

### check jks
D:\work\installed\java\bin\keytool -list -v -keystore ca.jks -storepass password
D:\work\installed\java\bin\keytool -list -v -keystore truststore.jks -storepass password

### export cert from jks
D:\work\installed\java\bin\keytool -export -keystore ca.jks -alias caKey -file ca.cer -storepass password

### print cert
D:\work\installed\java\bin\keytool -printcert -file ca.cer

### create jks from cert
D:\work\installed\java\bin\keytool -import -alias caKey -file ca.cer -keystore truststore.jks -storepass password
